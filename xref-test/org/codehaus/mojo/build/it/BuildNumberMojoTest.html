48      @Test
49      public void basicItTest()
50          throws Exception
51      {
52          Assume.assumeTrue( isSvn18() );
53  
54          File projDir = resources.getBasedir( "basic-it" );
55  
56          MavenExecution mavenExec = maven.forProject( projDir );
57          MavenExecutionResult result = mavenExec.execute( "clean" );
58          result.assertErrorFreeLog();
59          File testDir = result.getBasedir();
60          FileUtils.copyDirectoryStructure( new File( testDir, "dotSvnDir" ), new File( testDir, ".svn" ) );
61          result = mavenExec.execute( "clean", "verify" );
62          result.assertLogText( "Storing buildNumber: 14069" );
63          result.assertLogText( "Storing scmBranch: trunk" );
64  
65          File artifact = new File( testDir, "target/buildnumber-maven-plugin-basic-it-1.0-SNAPSHOT.jar" );
66          JarFile jarFile = new JarFile( artifact );
67          Attributes manifest = jarFile.getManifest().getMainAttributes();
68          jarFile.close();
69          String scmRev = manifest.getValue( "SCM-Revision" );
70          Assert.assertEquals( "14069", scmRev );
71      }

379     private static boolean isSvn18()
380     {
381         Commandline cl = new Commandline();
382         cl.setExecutable( "svn" );
383         cl.createArg().setValue( "--version" );
384 
385         StringStreamConsumer stdout = new StringStreamConsumer();
386         StringStreamConsumer stderr = new StringStreamConsumer();
387 
388         try
389         {
390             CommandLineUtils.executeCommandLine( cl, stdout, stderr );
391             Matcher versionMatcher = SVN_VERSION_PATTERN.matcher( stdout.getOutput() );
392             return versionMatcher.find() && ( Integer.parseInt( versionMatcher.group( 1 ) ) > 1
393                 || Integer.parseInt( versionMatcher.group( 2 ) ) >= 8 );
394         }
395         catch ( CommandLineException e )
396         {
397         }
398 
399         return false;
400     }
