18  @RunWith( MavenJUnitTestRunner.class )

20  public class CreateMetadataMojoTest
21  {
22      @Rule
23      public final TestResources resources = new TestResources();
24  
25      public final MavenRuntime maven;
26  
27      public CreateMetadataMojoTest( MavenRuntimeBuilder builder )
28          throws Exception
29      {
30          this.maven = builder.withCliOptions( "-B" ).build();
31      }
32  
33      @Test
34      public void testBasicConfiguration()
35          throws Exception
36      {
37          File projDir = resources.getBasedir( "create-metadata-it" );
38  
39          MavenExecution mavenExec = maven.forProject( projDir );
40          MavenExecutionResult result = mavenExec.execute( "clean", "test" );
41          result.assertErrorFreeLog();
42  
43          File testDir = result.getBasedir();
44          Assert.assertTrue( new File( testDir, "target/file1.properties" ).exists() );
45          Assert.assertTrue( new File( testDir, "target/xxx/file1.properties" ).exists() );
46          Assert.assertTrue( new File( testDir, "target/generated/build-metadata/build.properties" ).exists() );
47          Assert.assertTrue( new File( testDir, "target/classes/build.properties" ).exists() );
48  
49      }
