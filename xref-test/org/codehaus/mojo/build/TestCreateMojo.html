72      @Test
73      public void testMessageFormat()
74          throws Exception
75      {
76          CreateMojo mojo = new CreateMojo();
77          mojo.setFormat( "At {1,time} on {1,date}, there was {2} on planet {0,number,integer}." );
78          mojo.setItems( asList( 7, "timestamp", "a disturbance in the Force" ) );
79  
80          Locale currentLocale = Locale.getDefault();
81          try
82          {
83              Locale.setDefault( Locale.US );
84  
85              mojo.execute();
86  
87              String rev = mojo.getRevision();
88  
89              assertTrue( "Revision '" + rev + "' has wrong format.",
90                          rev.matches( "^At (\\d{1,2}:?){3} (AM|PM) on \\w{3} \\d{1,2}, \\d{4}, there was a disturbance in the Force on planet 7." ) );
91  
92          }
93          finally
94          {
95              Locale.setDefault( currentLocale );
96          }
97      }

102     @Test
103     public void testLocale()
104         throws Exception
105     {
106         Date date = new Date( 0 ); // the epoch
107         CreateMojo mojo = new CreateMojo();
108 
109         mojo.setFormat( "{0,date}" );
110         mojo.setItems( asList( date ) );
111 
112         mojo.execute();
113         assertEquals( DateFormat.getDateInstance( DateFormat.DEFAULT ).format( date ), mojo.getRevision() );
114 
115         mojo.setLocale( "en" );
116         mojo.execute();
117         SimpleDateFormat dateFormat = new SimpleDateFormat( "MMM dd, yyyy", new Locale( "en ") );
118         dateFormat.parse( mojo.getRevision() );
119         // assertEquals( "Jan 1, 1970", mojo.getRevision() );
120 
121         mojo.setLocale( "fi" );
122         mojo.execute();
123         dateFormat = new SimpleDateFormat( "dd.mm.yyyy" );
124         dateFormat.parse( mojo.getRevision() );
125         // assertEquals( "1.1.1970", mojo.getRevision() );
126 
127         mojo.setLocale( "de" );
128         mojo.execute();
129         dateFormat = new SimpleDateFormat( "dd.mm.yyyy" );
130         dateFormat.parse( mojo.getRevision() );
131         // assertEquals( "01.01.1970", mojo.getRevision() );
132     }

134     @Test
135     public void testSequenceFormat()
136         throws Exception
137     {
138         CreateMojo mojo = new CreateMojo();
139         mojo.setBuildNumberPropertiesFileLocation( new File( folder.getRoot(), "target/buildNumber.properties" ) );
140         mojo.setFormat( "{0,number}.{1,number}.{2,number}" );
141         mojo.setItems( asList( "buildNumber0", "buildNumber1", "buildNumber2" ) );
142 
143         File file = new File( folder.getRoot(), "target/buildNumber.properties" );
144         file.delete();
145 
146         mojo.execute();
147 
148         String rev = mojo.getRevision();
149 
150         assertTrue( "Revision '" + rev + "' has wrong format.", rev.matches( "(\\d+\\.?){3}" ) );
151 
152         assertTrue( file.exists() );
153     }

155     @Test
156     public void testFilterBranchFromScmUrl()
157     {
158         CreateMojo mojo = new CreateMojo();
159         String scmUrlTrunk = "https://mifos.dev.java.net/svn/mifos/trunk";
160         assertEquals( "trunk", mojo.filterBranchFromScmUrl( scmUrlTrunk ) );
161         String scmUrlBranch = "https://mifos.dev.java.net/svn/mifos/branches/v1.2.x";
162         assertEquals( "branches/v1.2.x", mojo.filterBranchFromScmUrl( scmUrlBranch ) );
163         String scmUrlTag = "https://mifos.dev.java.net/svn/mifos/tags/v1.2.1";
164         assertEquals( "tags/v1.2.1", mojo.filterBranchFromScmUrl( scmUrlTag ) );
165     }

167     @Test
168     public void testFilterBranchFromScmUrlWithSubFolder()
169     {
170         CreateMojo mojo = new CreateMojo();
171         String scmUrlTrunk = "https://mifos.dev.java.net/svn/mifos/trunk/subfolder";
172         assertEquals( "trunk", mojo.filterBranchFromScmUrl( scmUrlTrunk ) );
173         String scmUrlBranch = "https://mifos.dev.java.net/svn/mifos/branches/v1.2.x/subfolder";
174         assertEquals( "branches/v1.2.x", mojo.filterBranchFromScmUrl( scmUrlBranch ) );
175         String scmUrlTag = "https://mifos.dev.java.net/svn/mifos/tags/v1.2.1/subfolder";
176         assertEquals( "tags/v1.2.1", mojo.filterBranchFromScmUrl( scmUrlTag ) );
177     }
