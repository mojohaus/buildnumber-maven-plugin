24  import java.text.SimpleDateFormat;
25  import java.util.Calendar;
26  import java.util.Date;
27  import java.util.TimeZone;
28  
29  import org.apache.commons.lang.StringUtils;
30  
31  public class Utils
32  {
33      private Utils()
34      {
35      }
36  
37      public static String createTimestamp( String timestampFormat, String timeZoneId )
38      {
39          return createTimestamp( timestampFormat, timeZoneId, null );
40      }
41  
42      public static String createTimestamp( String timestampFormat, String timeZoneId, Date now )
43      {
44          if ( null == now )
45              now = Calendar.getInstance().getTime();
46  
47          if ( StringUtils.isBlank( timestampFormat ) )
48          {
49              return String.valueOf( now.getTime() );
50          }
51          else
52          {
53              SimpleDateFormat dateFormat = new SimpleDateFormat( timestampFormat );
54              dateFormat.setTimeZone( getTimeZone( timeZoneId ) );
55              return dateFormat.format( now );
56          }
57      }
58  
59      private static TimeZone getTimeZone( String timeZoneId )
60      {
61          TimeZone timeZone = TimeZone.getDefault();
62          if ( StringUtils.isNotBlank( timeZoneId ) )
63          {
64              timeZone = TimeZone.getTimeZone( timeZoneId );
65          }
66          return timeZone;
67      }
68  }
