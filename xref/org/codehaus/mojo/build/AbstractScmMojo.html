149     /**
150      * Load username password from settings.
151      */
152     private void loadInfosFromSettings( ScmProviderRepositoryWithHost repo )
153     {
154         if ( username == null || password == null )
155         {
156             String host = repo.getHost();
157 
158             int port = repo.getPort();
159 
160             if ( port > 0 )
161             {
162                 host += ":" + port;
163             }
164 
165             Server server = this.settings.getServer( host );
166 
167             if ( server != null )
168             {
169                 setPasswordIfNotEmpty( repo, decrypt( server.getPassword(), host ) );
170 
171                 setUserIfNotEmpty( repo, server.getUsername() );
172             }
173         }
174     }
175 
176     private String decrypt( String str, String server )
177     {
178         try
179         {
180             return securityDispatcher.decrypt( str );
181         }
182         catch ( SecDispatcherException e )
183         {
184             getLog().warn( "Failed to decrypt password/passphrase for server " + server + ", using auth token as is" );
185             return str;
186         }
187     }
188 
189     protected ScmRepository getScmRepository()
190         throws ScmException
191     {
192         ScmRepository repository = scmManager.makeScmRepository( !StringUtils.isBlank( this.scmConnectionUrl )
193                         ? scmConnectionUrl : scmDeveloperConnectionUrl );
194 
195         ScmProviderRepository scmRepo = repository.getProviderRepository();
196 
197         if ( scmRepo instanceof ScmProviderRepositoryWithHost )
198         {
199             loadInfosFromSettings( (ScmProviderRepositoryWithHost) scmRepo );
200         }
201 
202         setPasswordIfNotEmpty( scmRepo, password );
203 
204         setUserIfNotEmpty( scmRepo, username );
205 
206         return repository;
207     }
208 
209     private void setPasswordIfNotEmpty( ScmProviderRepository repository, String password )
210     {
211         if ( !StringUtils.isEmpty( password ) )
212         {
213             repository.setPassword( password );
214         }
215     }
216 
217     private void setUserIfNotEmpty( ScmProviderRepository repository, String user )
218     {
219         if ( !StringUtils.isEmpty( user ) )
220         {
221             repository.setUser( user );
222         }
223     }

225     protected void checkResult( ScmResult result )
226         throws ScmException
227     {
228         if ( !result.isSuccess() )
229         {
230             // TODO: improve error handling
231             getLog().error( "Provider message:" );
232 
233             getLog().error( result.getProviderMessage() );
234 
235             getLog().error( "Command output:" );
236 
237             getLog().error( result.getCommandOutput() );
238 
239             throw new ScmException( "Error!" );
240         }
241     }

246     /**
247      * Get info from scm.
248      *
249      * @param repository
250      * @param fileSet
251      * @return
252      * @throws ScmException
253      */
254     protected InfoScmResult info( ScmRepository repository, ScmFileSet fileSet )
255         throws ScmException
256     {
257         CommandParameters commandParameters = new CommandParameters();
258 
259         // only for Git, we will make a test for shortRevisionLength parameter
260         if ( GitScmProviderRepository.PROTOCOL_GIT.equals( scmManager.getProviderByRepository( repository ).getScmType() )
261             && this.shortRevisionLength > 0 )
262         {
263             getLog().info( "ShortRevision tag detected. The value is '" + this.shortRevisionLength + "'." );
264             if ( shortRevisionLength >= 0 && shortRevisionLength < 4 )
265             {
266                 getLog().warn( "shortRevision parameter less then 4. ShortRevisionLength is relaying on 'git rev-parese --short=LENGTH' command, accordingly to Git rev-parse specification the LENGTH value is miminum 4. " );
267             }
268             commandParameters.setInt( CommandParameter.SCM_SHORT_REVISION_LENGTH, this.shortRevisionLength );
269         }
270 
271         if ( !StringUtils.isBlank( scmTag ) && !"HEAD".equals( scmTag ) )
272         {
273             commandParameters.setScmVersion( CommandParameter.SCM_VERSION, new ScmTag( scmTag ) );
274         }
275 
276         return scmManager.getProviderByRepository( repository ).info( repository.getProviderRepository(), fileSet,
277                                                                       commandParameters );
278     }
279 
280     protected String getScmRevision()
281         throws ScmException
282     {
283         ScmRepository repository = getScmRepository();
284 
285         InfoScmResult scmResult = info( repository, new ScmFileSet( scmDirectory ) );
286 
287         if ( scmResult == null || scmResult.getInfoItems().isEmpty() )
288         {
289             return ( !StringUtils.isEmpty( revisionOnScmFailure ) ) ? revisionOnScmFailure : null;
290         }
291 
292         checkResult( scmResult );
293 
294         InfoItem info = scmResult.getInfoItems().get( 0 );
295 
296         if ( useLastCommittedRevision )
297         {
298             // It seemed the case that for git getLastChangedRevision()
299             // returns null instead of the last revision in contradiction
300             // to the above code: scmResult.getInfoItems().get(0);
301             return info.getLastChangedRevision();
302         }
303 
304         return info.getRevision();
305     }
