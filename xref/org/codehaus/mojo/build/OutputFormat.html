24  import java.io.IOException;
25  import java.io.OutputStream;
26  import java.util.Properties;
27  
28  public abstract class OutputFormat
29  {
30      static final OutputFormat DEFAULT_FORMAT = new PropertiesOutputFormat();
31  
32      private static final OutputFormat[] FORMATS =
33          new OutputFormat[] { new JsonOutputFormat(), OutputFormat.DEFAULT_FORMAT };
34  
35      public static OutputFormat getOutputFormatFor( String fileName )
36      {
37          for ( OutputFormat outputFormat : OutputFormat.FORMATS )
38          {
39              if ( outputFormat.handles( fileName ) )
40              {
41                  return outputFormat;
42              }
43          }
44  
45          return OutputFormat.DEFAULT_FORMAT;
46      }
47  
48      public abstract boolean handles( String fileName );
49  
50      public abstract void write( Properties props, OutputStream out )
51          throws IOException;
52  }
