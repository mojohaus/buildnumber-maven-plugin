24  import java.io.File;
25  
26  import org.apache.maven.plugin.AbstractMojo;
27  import org.apache.maven.plugin.MojoExecutionException;
28  import org.apache.maven.plugins.annotations.LifecyclePhase;
29  import org.apache.maven.plugins.annotations.Mojo;
30  import org.apache.maven.plugins.annotations.Parameter;
31  import org.apache.maven.project.MavenProject;
32  import org.apache.maven.scm.ScmException;
33  import org.apache.maven.scm.ScmResult;
34  import org.apache.maven.scm.log.ScmLogDispatcher;
35  import org.apache.maven.scm.provider.hg.HgUtils;
36  import org.codehaus.plexus.util.StringUtils;
37  
38  /**
39   * Goal which sets project properties for changeSet and changeSetDate from the current Mercurial repository.
40   *
41   * @author Tomas Pollak
42   * @since 1.0-beta-4
43   */
44  @Mojo( name = "hgchangeset", defaultPhase = LifecyclePhase.INITIALIZE, requiresProject = true, threadSafe = true )
45  public class HgChangeSetMojo
46      extends AbstractMojo
47  {
48      /**
49       * Whether to skip this execution.
50       *
51       * @since 1.3
52       */
53      @Parameter( property = "maven.buildNumber.skip", defaultValue = "false" )
54      private boolean skip;
55  
56      private ScmLogDispatcher logger = new ScmLogDispatcher();
57  
58      /**
59       * The maven project.
60       */
61      @Parameter( defaultValue = "${project}", required = true, readonly = true )
62      private MavenProject project;
63  
64      /**
65       * Local directory to be used to issue SCM actions
66       *
67       * @since 1.0
68       */
69      @Parameter( property = "maven.changeSet.scmDirectory", defaultValue = "${basedir}" )
70      private File scmDirectory;
71  
72      /**
73       * Forces to use last logged changeset/changesetDate from scmDirectory and not current repository
74       * changeset/changesetDate.
75       *
76       * @since 1.5
77       */
78      @Parameter( property = "maven.buildNumber.useLastChangeSetInDirectory", defaultValue = "false" )
79      private Boolean useLastChangeSetInDirectory;
80  
81      private void checkResult( ScmResult result )
82          throws MojoExecutionException
83      {
84          if ( !result.isSuccess() )
85          {
86              getLog().debug( "Provider message:" );
87              getLog().debug( result.getProviderMessage() == null ? "" : result.getProviderMessage() );
88              getLog().debug( "Command output:" );
89              getLog().debug( result.getCommandOutput() == null ? "" : result.getCommandOutput() );
90              throw new MojoExecutionException( "Command failed."
91                  + StringUtils.defaultString( result.getProviderMessage() ) );
92          }
93      }
94  
95      public void execute()
96          throws MojoExecutionException
97      {
98          if ( skip )
99          {
100             getLog().info( "Skipping execution." );
101             return;
102         }
103 
104         try
105         {
106             String previousChangeSet = getChangeSetProperty();
107             String previousChangeSetDate = getChangeSetDateProperty();
108             if ( previousChangeSet == null || previousChangeSetDate == null )
109             {
110                 String changeSet = getChangeSet();
111                 String changeSetDate = getChangeSetDate();
112                 getLog().info( "Setting Mercurial Changeset: " + changeSet );
113                 getLog().info( "Setting Mercurial Changeset Date: " + changeSetDate );
114                 setChangeSetProperty( changeSet );
115                 setChangeSetDateProperty( changeSetDate );
116             }
117         }
118         catch ( ScmException e )
119         {
120             throw new MojoExecutionException( "SCM Exception", e );
121         }
122     }
123 
124     protected String getHgCommandOutput( String[] command )
125         throws ScmException, MojoExecutionException
126     {
127         HgOutputConsumer consumer = new HgOutputConsumer( logger );
128         ScmResult result = HgUtils.execute( consumer, logger, scmDirectory, command );
129         checkResult( result );
130         return consumer.getOutput();
131     }
132 
133     protected String getChangeSet()
134         throws ScmException, MojoExecutionException
135     {
136         return getHgCommandOutput( useLastChangeSetInDirectory
137                         ? new String[] { "log", "-l1", "--template", "\"{node|short}\"", "." }
138                         : new String[] { "id", "-i" } );
139     }
140 
141     protected String getChangeSetDate()
142         throws ScmException, MojoExecutionException
143     {
144         return getHgCommandOutput( useLastChangeSetInDirectory
145                         ? new String[] { "log", "-l1", "--template", "\"{date|isodate}\"", "." }
146                         : new String[] { "log", "-r", ".", "--template", "\"{date|isodate}\"" } );
147     }
148 
149     protected String getChangeSetDateProperty()
150     {
151         return getProperty( "changeSetDate" );
152     }
153 
154     protected String getChangeSetProperty()
155     {
156         return getProperty( "changeSet" );
157     }
158 
159     protected String getProperty( String property )
160     {
161         return project.getProperties().getProperty( property );
162     }
163 
164     private void setChangeSetDateProperty( String changeSetDate )
165     {
166         setProperty( "changeSetDate", changeSetDate );
167     }
168 
169     private void setChangeSetProperty( String changeSet )
170     {
171         setProperty( "changeSet", changeSet );
172     }
173 
174     private void setProperty( String property, String value )
175     {
176         if ( value != null )
177         {
178             project.getProperties().put( property, value );
179         }
180     }
