24  import java.io.IOException;
25  import java.io.OutputStream;
26  import java.util.Properties;
27  
28  public class PropertiesOutputFormat
29      extends OutputFormat
30  {
31      @Override
32      public boolean handles( String fileName )
33      {
34          return fileName.endsWith( ".properties" );
35      }
36  
37      @Override
38      public void write( Properties props, OutputStream out )
39          throws IOException
40      {
41          props.store( out, "Created by build system. Do not modify" );
42      }
43  }
