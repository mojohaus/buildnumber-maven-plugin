70  /**
71   * This mojo is designed to give you a build number. So when you might make 100 builds of version 1.0-SNAPSHOT, you can
72   * differentiate between them all.
73   * <p>
74   * The build number is based on the revision number retrieved from SCM. It is known to work with Subversion, GIT, and
75   * Mercurial.
76   * <p>
77   * This mojo can also check to make sure that you have checked everything into SCM, before issuing the build number.
78   * That behaviour can be suppressed, and then the latest local build number is used.
79   * <p>
80   * Build numbers are not automatically reflected in your artifact's filename, but can be added to the metadata. You can
81   * access the build number in your pom with ${buildNumber}. You can also access ${timestamp} and the SCM branch of the
82   * build (if applicable) in ${scmBranch}
83   * <p>
84   * Note that there are several <code><strong>doFoo</strong></code> parameters. These parameters (doCheck, doUpdate, etc)
85   * are the first thing evaluated. If there is no matching expression, we get the default-value. If there is (ie
86   * <code>-Dmaven.buildNumber.doUpdate=false</code>), we get that value. So if the XML contains
87   * <code>&lt;doCheck&gt;true&lt;/doCheck&gt;</code>, then normally that's the final value of the param in question. However,
88   * this mojo reverses that behaviour, such that the command line parameters get the last say.
89   *
90   * @author <a href="mailto:woodj@ucalgary.ca">Julian Wood</a>
91   */
92  @Mojo( name = "create", defaultPhase = LifecyclePhase.INITIALIZE, requiresProject = true, threadSafe = true )
93  public class CreateMojo
94      extends AbstractScmMojo

249     public void execute()
250         throws MojoExecutionException, MojoFailureException
251     {
252         String buildIsTainted = "ok";
253         
254         if ( skip )
255         {
256             getLog().info( "Skipping execution." );
257             return;
258         }
259 
260         if ( providerImplementations != null )
261         {
262             changeProviderImplementation();
263         }
264         Date now = Calendar.getInstance().getTime();
265         if ( format != null )
266         {
267             if ( items == null )
268             {
269                 throw new MojoExecutionException( " if you set a format, you must provide at least one item, "
270                     + "please check documentation " );
271             }
272             Object[] itemAry = handleItems( now );
273 
274             revision = format( itemAry );
275         }
276         else
277         {
278             // Check if the plugin has already run.
279             revision = project.getProperties().getProperty( this.buildNumberPropertyName );
280             if ( this.getRevisionOnlyOnce && revision != null )
281             {
282                 getLog().debug( "Revision available from previous execution" );
283                 return;
284             }
285 
286             if ( doCheck )
287             {
288                 StringBuilder message = new StringBuilder();
289                 if ( doLocalModificationExist( message ) )
290                 {
291                     buildIsTainted = "tainted";
292                     if ( failTheBuild )
293                     {
294                         throw new MojoExecutionException( "Cannot create the build number because you have local modifications : \n"
295                             + message );
296 
297                     }
298                 }
299             }
300             else
301             {
302                 getLog().debug( "Checking for local modifications: skipped." );
303             }
304 
305             if ( session.getSettings().isOffline() )
306             {
307                 getLog().info( "maven is executed in offline mode, Updating project files from SCM: skipped." );
308             }
309             else
310             {
311                 if ( doUpdate )
312                 {
313                     // we update your local repo
314                     // even after you commit, your revision stays the same until you update, thus this
315                     // action
316                     List<ScmFile> changedFiles = update();
317                     for ( ScmFile file : changedFiles )
318                     {
319                         getLog().debug( "Updated: " + file );
320                     }
321                     if ( changedFiles.size() == 0 )
322                     {
323                         getLog().debug( "No files needed updating." );
324                     }
325                 }
326                 else
327                 {
328                     getLog().debug( "Updating project files from SCM: skipped." );
329                 }
330             }
331             revision = getRevision();
332         }
333 
334         if ( project != null )
335         {
336             buildNumberAndTimeStampForReactorProjects( now, buildIsTainted );
337         }
338     }

340     private Object[] handleItems( Date now )
341         throws MojoExecutionException
342     {
343         // needs to be an array
344         // look for special values
345         Object[] itemAry = new Object[items.size()];
346         for ( int i = 0; i < items.size(); i++ )
347         {
348             Object item = items.get( i );
349             if ( item instanceof String )
350             {
351                 String s = (String) item;
352                 if ( s.equals( "timestamp" ) )
353                 {
354                     itemAry[i] = now;
355                 }
356                 else if ( s.startsWith( "scmVersion" ) )
357                 {
358                     useScm = true;
359                     itemAry[i] = getRevision();
360                 }
361                 else if ( s.startsWith( "buildNumber" ) )
362                 {
363                     // check for properties file
364                     File propertiesFile = this.buildNumberPropertiesFileLocation;
365 
366                     createPropertiesFileIfNotExists( propertiesFile );
367 
368                     Properties properties = new Properties();
369                     String buildNumberString = null;
370                     try (FileInputStream inputStream = new FileInputStream( propertiesFile ))
371                     {
372                         // get the number for the buildNumber specified
373                         properties.load( inputStream );
374                         buildNumberString = properties.getProperty( s );
375                         if ( buildNumberString == null )
376                         {
377                             buildNumberString = "0";
378                         }
379                         int buildNumber = parseInt( buildNumberString );
380 
381                         // store the increment
382                         properties.setProperty( s, String.valueOf( ++buildNumber ) );
383                         try (FileOutputStream outputStream = new FileOutputStream( propertiesFile )) {
384                             properties.store(outputStream, "maven.buildNumber.plugin properties file");
385                         }
386 
387                         // use in the message (format)
388                         itemAry[i] = buildNumber;
389                     }
390                     catch ( NumberFormatException e )
391                     {
392                         throw new MojoExecutionException( "Couldn't parse buildNumber in properties file to an Integer: "
393                             + buildNumberString );
394                     }
395                     catch ( IOException e )
396                     {
397                         throw new MojoExecutionException( "Couldn't load properties file: " + propertiesFile, e );
398                     }
399                 }
400                 else
401                 {
402                     itemAry[i] = item;
403                 }
404             }
405             else
406             {
407                 itemAry[i] = item;
408             }
409         }
410         return itemAry;
411     }

413     private void createPropertiesFileIfNotExists( File propertiesFile )
414         throws MojoExecutionException
415     {
416         // create if not exists
417         if ( !propertiesFile.exists() )
418         {
419             try
420             {
421                 if ( !propertiesFile.getParentFile().exists() )
422                 {
423                     propertiesFile.getParentFile().mkdirs();
424                 }
425                 propertiesFile.createNewFile();
426             }
427             catch ( IOException e )
428             {
429                 throw new MojoExecutionException( "Couldn't create properties file: " + propertiesFile, e );
430             }
431         }
432     }

434     private void buildNumberAndTimeStampForReactorProjects( Date now, String taintedValue )
435         throws MojoExecutionException
436     {
437         String timestamp = String.valueOf( now.getTime() );
438         if ( timestampFormat != null )
439         {
440             if ( timestampFormat.matches( "\\{0,date,[^\\}]+\\}" ) )
441             {
442                 getLog().warn( "The timestampFormat parameter now uses java.text.SimpleDateFormat." );
443                 getLog().warn( "Please update your POM as support for java.text.MessageFormat may be removed." );
444                 timestamp = Utils.createTimestamp( timestampFormat.replaceFirst( "\\{0,date,([^\\}]+)\\}", "$1" ),
445                                                    timezone, now );
446             }
447             else
448             {
449                 timestamp = Utils.createTimestamp( timestampFormat, timezone, now );
450             }
451         }
452 
453         getLog().info( MessageFormat.format( "Storing buildNumber: {0} at timestamp: {1}",
454                 revision, timestamp) );
455         if ( revision != null )
456         {
457             project.getProperties().put( buildNumberPropertyName, revision );
458         }
459         project.getProperties().put( timestampPropertyName, timestamp );
460 
461         String scmBranch = getScmBranch();
462         getLog().info( "Storing buildScmBranch: " + scmBranch );
463         project.getProperties().put( scmBranchPropertyName, scmBranch );
464 
465         // Add the revision and timestamp properties to each project in the reactor
466         if ( getRevisionOnlyOnce && reactorProjects != null )
467         {
468             Iterator<MavenProject> projIter = reactorProjects.iterator();
469             while ( projIter.hasNext() )
470             {
471                 MavenProject nextProj = projIter.next();
472                 if ( revision != null )
473                 {
474                     nextProj.getProperties().put( this.buildNumberPropertyName, revision );
475                 }
476                 nextProj.getProperties().put( this.timestampPropertyName, timestamp );
477                 nextProj.getProperties().put( this.scmBranchPropertyName, scmBranch );
478                 nextProj.getProperties().put( this.buildTaintedPropertyName, taintedValue );
479             }
480         }
481     }

483     private void changeProviderImplementation()
484     {
485         for ( Entry<String, String> entry : providerImplementations.entrySet() )
486         {
487             String providerType = entry.getKey();
488             String providerImplementation = entry.getValue();
489             getLog().info( "Change the default '" + providerType + "' provider implementation to '"
490                 + providerImplementation + "'." );
491             scmManager.setScmProviderImplementation( providerType, providerImplementation );
492         }

495     /**
496      * Formats the given argument using the configured format template and locale.
497      *
498      * @param arguments arguments to be formatted @ @return formatted result
499      */
500     private String format( Object[] arguments )
501     {
502         Locale l = Locale.getDefault();
503         if ( locale != null )
504         {
505             String[] parts = locale.split( "_", 3 );
506             if ( parts.length <= 1 )
507             {
508                 l = new Locale( locale );
509             }
510             else if ( parts.length == 2 )
511             {
512                 l = new Locale( parts[0], parts[1] );
513             }
514             else
515             {
516                 l = new Locale( parts[0], parts[1], parts[2] );
517             }
518         }
519 
520         return new MessageFormat( format, l ).format( arguments );
521     }

523     private boolean doLocalModificationExist( StringBuilder message )
524         throws MojoExecutionException
525     {
526         boolean result = false;
527 
528         getLog().debug( "Verifying there are no local modifications ..." );
529 
530         List<ScmFile> changedFiles;
531 
532         try
533         {
534             changedFiles = getStatus();
535         }
536         catch ( ScmException e )
537         {
538             throw new MojoExecutionException( "An error has occurred while checking scm status.", e );
539         }
540 
541         if ( !changedFiles.isEmpty() )
542         {
543             for ( ScmFile file : changedFiles )
544             {
545                 message.append( file.toString() );
546 
547                 message.append( System.lineSeparator() );
548             }
549 
550             result = true;
551         }
552 
553         return result;
554     }

556     public List<ScmFile> update()
557         throws MojoExecutionException
558     {
559         try
560         {
561             ScmRepository repository = getScmRepository();
562 
563             ScmProvider scmProvider = scmManager.getProviderByRepository( repository );
564 
565             UpdateScmResult result = scmProvider.update( repository, new ScmFileSet( scmDirectory ) );
566 
567             if ( result == null )
568             {
569                 return Collections.emptyList();
570             }
571 
572             checkResult( result );
573 
574             if ( result instanceof UpdateScmResultWithRevision )
575             {
576                 String revision = ( (UpdateScmResultWithRevision) result ).getRevision();
577                 getLog().info( "Got a revision during update: " + revision );
578                 this.revision = revision;
579             }
580 
581             return result.getUpdatedFiles();
582         }
583         catch ( ScmException e )
584         {
585             throw new MojoExecutionException( "Couldn't update project. " + e.getMessage(), e );
586         }
587 
588     }

590     public List<ScmFile> getStatus()
591         throws ScmException
592     {
593 
594         ScmRepository repository = getScmRepository();
595 
596         ScmProvider scmProvider = scmManager.getProviderByRepository( repository );
597 
598         StatusScmResult result = scmProvider.status( repository, new ScmFileSet( scmDirectory ) );
599 
600         if ( result == null )
601         {
602             return Collections.emptyList();
603         }
604 
605         checkResult( result );
606 
607         return result.getChangedFiles();
608

611     /**
612      * Get the branch info for this revision from the repository. For svn, it is in svn info.
613      *
614      * @return
615      * @throws MojoExecutionException
616      * @throws MojoExecutionException
617      */
618     public String getScmBranch()
619         throws MojoExecutionException
620     {
621         try
622         {
623             ScmRepository repository = getScmRepository();
624             ScmProvider provider = scmManager.getProviderByRepository( repository );
625             /* git branch can be obtained directly by a command */
626             if ( GitScmProviderRepository.PROTOCOL_GIT.equals( provider.getScmType() ) )
627             {
628                 ScmFileSet fileSet = new ScmFileSet( scmDirectory );
629                 return GitBranchCommand.getCurrentBranch( getLogger(),
630                                                           (GitScmProviderRepository) repository.getProviderRepository(),
631                                                           fileSet );
632             }
633             else if ( provider instanceof HgScmProvider )
634             {
635                 /* hg branch can be obtained directly by a command */
636                 HgOutputConsumer consumer = new HgOutputConsumer( getLogger() );
637                 ScmResult result = HgUtils.execute( consumer, logger, scmDirectory, new String[] { "id", "-b" } );
638                 checkResult( result );
639                 if ( StringUtils.isNotEmpty( consumer.getOutput() ) )
640                 {
641                     return consumer.getOutput();
642                 }
643             }
644         }
645         catch ( ScmException e )
646         {
647             getLog().warn( "Cannot get the branch information from the git repository: \n" + e.getLocalizedMessage() );
648         }
649 
650         return getScmBranchFromUrl();
651     }

653     private String getScmBranchFromUrl()
654         throws MojoExecutionException
655     {
656         String scmUrl = null;
657         try
658         {
659             ScmRepository repository = getScmRepository();
660             InfoScmResult scmResult = info( repository, new ScmFileSet( scmDirectory ) );
661             if ( scmResult == null || !scmResult.isSuccess() )
662             {
663                 getLog().debug( "Cannot get the branch information from the scm repository : "
664                     + ( scmResult == null ? "" : scmResult.getCommandOutput() ) );
665                 return DEFAULT_BRANCH_NAME;
666             }
667             if ( scmResult.getInfoItems().isEmpty() )
668             {
669                 if ( !StringUtils.isEmpty( revisionOnScmFailure ) )
670                 {
671                     setDoCheck( false );
672                     setDoUpdate( false );
673 
674                     return DEFAULT_BRANCH_NAME;
675                 }
676             }
677             if ( !scmResult.getInfoItems().isEmpty() )
678             {
679                 InfoItem info = scmResult.getInfoItems().get( 0 );
680                 scmUrl = info.getURL();
681             }
682         }
683         catch ( ScmException e )
684         {
685             if ( !StringUtils.isEmpty( revisionOnScmFailure ) )
686             {
687                 getLog().warn( "Cannot get the branch information from the scm repository, proceeding with "
688                     + DEFAULT_BRANCH_NAME + " : \n" + e.getLocalizedMessage() );
689 
690                 setDoCheck( false );
691                 setDoUpdate( false );
692 
693                 return DEFAULT_BRANCH_NAME;
694             }
695             throw new MojoExecutionException( "Cannot get the branch information from the scm repository : \n"
696                 + e.getLocalizedMessage(), e );
697         }
698 
699         return filterBranchFromScmUrl( scmUrl );
700     }
701 
702     protected String filterBranchFromScmUrl( String scmUrl )
703     {
704         String scmBranch = "UNKNOWN";
705 
706         if ( StringUtils.contains( scmUrl, "/trunk" ) )
707         {
708             scmBranch = "trunk";
709         }
710         else if ( StringUtils.contains( scmUrl, "/branches" ) || StringUtils.contains( scmUrl, "/tags" ) )
711         {
712             scmBranch = scmUrl.replaceFirst( ".*((branches|tags)/[^/]*).*", "$1" );
713         }
714         return scmBranch;
715     }

718      * Get the revision info from the repository. For svn, it is svn info
719      *
720      * @return
721      * @throws MojoExecutionException
722      */
723     public String getRevision()
724         throws MojoExecutionException
725     {
726 
727         if ( format != null && !useScm )
728         {
729             return revision;
730         }
731         useScm = false;
732 
733         try
734         {
735             return this.getScmRevision();
736         }
737         catch ( ScmException e )
738         {
739             if ( !StringUtils.isEmpty( revisionOnScmFailure ) )
740             {
741                 getLog().warn( "Cannot get the revision information from the scm repository, proceeding with "
742                     + "revision of " + revisionOnScmFailure + " : \n" + e.getLocalizedMessage() );
743 
744                 setDoCheck( false );
745                 setDoUpdate( false );
746 
747                 return revisionOnScmFailure;
748             }
749 
750             throw new MojoExecutionException( "Cannot get the revision information from the scm repository : \n"
751                 + e.getLocalizedMessage(), e );
752 
753         }
754 
755     }

802     private boolean getBooleanProperty( String key, boolean defaultValue )
803     {
804         String systemProperty = System.getProperty( key );
805         if ( systemProperty == null )
806         {
807             return defaultValue;
808         }
809         else
810         {
811             return parseBoolean( systemProperty );
812         }
