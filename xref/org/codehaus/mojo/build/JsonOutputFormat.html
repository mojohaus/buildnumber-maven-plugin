1   package org.codehaus.mojo.build;
2   
3   import com.google.gson.Gson;
4   import com.google.gson.stream.JsonWriter;
5   
6   import java.io.*;
7   import java.nio.charset.StandardCharsets;
8   import java.util.Properties;
9   
10  public class JsonOutputFormat
11      extends OutputFormat
12  {
13      @Override
14      public boolean handles( String fileName )
15      {
16          return fileName.endsWith( ".json" );
17      }
18  
19      @Override
20      public void write( Properties props, OutputStream out )
21          throws IOException
22      {
23          Gson gson = new Gson();
24          try (JsonWriter jsonWriter = gson.newJsonWriter( new OutputStreamWriter( out, StandardCharsets.UTF_8) )) {
25              jsonWriter.beginObject();
26              for (Object key : props.keySet()) {
27                  jsonWriter.name((String) key);
28                  jsonWriter.value(props.getProperty((String) key));
29              }
30              jsonWriter.endObject();
31              jsonWriter.flush();
32          }
33      }
34  }
